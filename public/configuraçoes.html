'Saudável' : 'Instável', latency: `${latency}ms` }; } catch (err) { return { status: 'Desconectado', latency: '∞' }; } } } // 12. COMPONENTE: PAINEL DE SAÚDE (HEALTH PANEL COMPONENT) const HealthPanel = { async render() { const health = await DiagnosticEngine.checkConnection(); return `

Saúde da Árvore

Status: ${health.status} Latência: ${health.latency}
`; } }; // 13. INTEGRAÇÃO DO PAINEL DE SAÚDE window.addEventListener('load', async () => { const main = document.querySelector('main'); const healthHtml = await HealthPanel.render(); main.insertAdjacentHTML('beforeend', healthHtml); }); // 14.
